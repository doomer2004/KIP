s = float(input('Введіть число: ')) #вводимо число виду nnn.ddd
a = (round(float(s) - int(s), 3)) * 1000 + int(s) / 1000 #формула для знаходження оберненого, ціле число x ділимо на 1000(ддя отримання числа виду nnn), а x таке, що має цифри після коми віднамаємо від цілого x, мнодимо на 1000 та знаходимо знвчення виду ddd
print(a) #виводимо число виду ddd.nnn
